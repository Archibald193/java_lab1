

    public void deposit(double n) {
        if (n > 0) {
            this.balance += n;
            System.out.println("На счет " + acNum + " внесено " + n + ". Новый баланс: " + this.balance);
        }
    }
    
    public void withdraw(double n) {
        if (n > 0 && this.balance >= n) {
            this.balance -= n;
            System.out.println("Со счета " + acNum + " снято " + n + ". Новый баланс: " + this.balance);
        } else {
             System.out.println("Недостаточно средств на счете " + acNum);
        }
    }
